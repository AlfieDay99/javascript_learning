    console.log("Regular Function:", this);

ðŸ”¹ Closures

Loop with setTimeout

// Using var (problematic)
for (var i = 0; i < 3; i++) {
    setTimeout(function() {
        console.log("var loop:", i);
    }, 1000);
}

// Using let (correct)
for (let j = 0; j < 3; j++) {
    setTimeout(function() {
        console.log("let loop:", j);
    }, 1000);
}

// Using closure with var
for (var i = 0; i < 3; i++) {
    (function(x) {
        setTimeout(function() {
            console.log("closure with var:", x);
        }, 1000);
    })(i);
}
